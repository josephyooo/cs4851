def sample_mnist(n_train=2000, n_test=1000):
    trX, trY, tsX, tsY = get_mnist()
    #Shuffle the data.
    tr_idx = np.random.choice(trX.shape[0], n_train)
    trX = trX[tr_idx]
    trY = trY[tr_idx]
    ts_idx = np.random.choice(tsX.shape[0], n_test)
    tsX = tsX[ts_idx]
    tsY = tsY[ts_idx]
    # Reshape to [n x m]. n dimensions and m samples
    trX = trX.reshape(-1, 28*28).T
    trY = trY.reshape(1, -1)
    tsX = tsX.reshape(-1, 28*28).T
    tsY = tsY.reshape(1, -1)
    return trX, trY, tsX, tsY

                self.stats['bnorm_running_mean'+str(ly+1)] = torch.zeros((n,1))
                self.stats['bnorm_running_var'+str(ly+1)] = torch.zeros((n,1))

                if apply_momentum:

                    self.stats['Gbnorm_beta'+str(ly+1)] = torch.zeros((n,1))
                    self.stats['Vbnorm_gamma'+str(ly+1)] = torch.zeros((n,1))
                    self.stats['Gbnorm_gamma'+str(ly+1)] = torch.zeros((n,1))


    ## Function to implement batchnorm ## 10 Points ##
    def batchnorm(self, A, ly, mode='train'):
        '''
        Batchnorm normalizes the input x to mean beta and standard deviation gamma

        Inputs:
            A: Input after activation - shape is (n,m), m samples where each sample is (n,1)
            ly: Layer index
            mode = 'train' or 'test'

        Outputs:
            Anorm: Normalized version of input x - shape (n,m)
        '''
        # When there is no batch norm for a layer
        if self.bnorm_list[ly] == 0:
            return A

        #beta: mean vector which will be the center of the data to beta after batchnorm - shape is (n,1)
        beta = self.parameters['bnorm_beta'+str(ly+1)]
        #gamma: standard deviation vector which will be scale of the data after batchnorm - shape (n,1)
        gamma = self.parameters['bnorm_gamma'+str(ly+1)]

        # epsilon value used for scaling during normalization to avoid divide by zero.
        epsilon = 1e-5
        # Exponential Weighted Averaging for Running Batch Mean and Variance
        beta_batch_stats = 0.9

        if mode == 'train':
            ### BEGIN SOLUTION ###
            B = A.shape[0]
            mu = 1/B * torch.sum(A, 1).reshape(-1, 1)
            var = 1/B * torch.sum(torch.square(A-mu))
            # Save the batch mean 'mu' and the batch var using exponential averaging where beta is beta_batch_stats
            self.stats['bnorm_running_mean'+str(ly+1)] *= beta_batch_stats
            self.stats['bnorm_running_mean'+str(ly+1)] += (1-beta_batch_stats)*mu
            self.stats['bnorm_running_var'+str(ly+1)] *= beta_batch_stats
            self.stats['bnorm_running_var'+str(ly+1)] += (1-beta_batch_stats)*var
        else:
            # Use the running batch mean and variance during test
            mu = self.stats['bnorm_running_mean'+str(ly+1)]
            var = self.stats['bnorm_running_var'+str(ly+1)]

        A -= mu
        A /= torch.sqrt(var+epsilon)
        A *= gamma
        A += beta
        ### END SOLUTION ###
        return A

    ## Function to implement dropout ## 10 points ##
    def dropout(self, A, ly, mode='train'):
        '''
        Using the 'inverted dropout' technique to implement dropout regularization.
        Inputs:
            A: Activation input before dropout is applied - shape is (n,m)
            drop_prob: dropout parameter. If drop_prob = 0.3, we drop 30% of the neuron activations
            mode: Dropout acts differently in training and testing mode. Hence, mode is a parameter which
                takes in only 2 values, 'train' or 'test'

        Outputs:
            A: Output of shape (n,m), with some values masked out and other values scaled to account for missing values
            cache: a tuple which stores the drop_prob, mode and mask for use in backward pass.
        '''
        # When there is no dropout return the same activation
        drop_prob = drop_prob_list[ly]
        mask = None
        if drop_prob == 0 or mode == 'test':
            return A

        # The prob_keep is the percentage of activations remaining after dropout
        # if drop_out = 0.3, then prob_keep = 0.7, i.e., 70% of the activations are retained
        prob_keep = 1-drop_prob

        # Note: instead of a binary mask implement a scaled mask, where mask is scaled by dividing it
        # by the prob_keep. For example, if we have input activations of size (3,4), then the mask is
        # mask = (torch.rand(3,4)<prob_keep)/prob_keep
        # We perform the scaling by prob_keep here so we don't have to do it specifically during backpropagation
        # We then update A by multiplying it element wise with the mask

        ### BEGIN SOLUTION ###
        mask = (torch.rand(*A.shape)<prob_keep)/prob_keep
        A *= mask
        ### END SOLUTION ###
        return A

    ## Forward propagation of input data through the network (No Softmax at the end) ## 10 Points ##
    def forward(self, x, mode='train'):
        for ly in range(self.num_layers):
            ## Apply the dot product followed by activation, batchnorm and dropout to get the logits
            ## Use torch.mm() for dot product
            ## You can use PyTorch relu or the relu() provided here
            ## Use your implementation for batchnorm and dropout
            ## Note: The last layer does not have an activation.
            ## Softmax and Cross-entropy are implemented together in the loss function F.cross_entropy()
            # But don't call F.cross_entropy here
            ### BEGIN SOLUTION ###
            x = torch.mm(self.parameters['W'+str(ly+1)], x) + self.parameters['b'+str(ly+1)]
            if ly < self.num_layers - 1:
                x = eval(self.activation_list[ly])(x)
                x = self.batchnorm(x, ly, mode)
                x = self.dropout(x, ly, mode)
            ### END SOLUTION ###
        return x

    #function to update parameters without momentum
    def update_parameters(self, gradients, alpha):
        '''
        Updates the network parameters with regular gradient descent

        Inputs:
            gradients: list of gradients for all the parameters in the network
                        The order of the list is the same as self.parameters.keys()
            alpha: The step_size for gradient descent

        Outputs:
            No outputs
        '''
        #convert gradients to a dictionary for easy access
        gradients = dict(zip(self.parameters.keys(), gradients))

        for ly in range(self.num_layers):
            # weights and bias update
            self.parameters["W"+str(ly+1)] -= alpha * gradients["W"+str(ly+1)]
            self.parameters["b"+str(ly+1)] -= alpha * gradients["b"+str(ly+1)]

            # batch norm parameters update
            if self.bnorm_list[ly]:
                self.parameters['bnorm_beta' + str(ly+1)] -= alpha * gradients["bnorm_beta"+str(ly+1)]
                self.parameters['bnorm_gamma' + str(ly+1)] -= alpha * gradients["bnorm_gamma"+str(ly+1)]


    #function to update parameters with momentum ## 20 points ##
    def update_parameters_with_momentum_Adam(self, gradients, alpha, beta=0.9, beta2=0.99, eps=1e-8):
        '''
        Updates the network parameters with Adam momentum

        Inputs:
            gradients: list of gradients for all the parameters in the network
                        The order of the list is the same as self.parameters.keys()
            alpha: The learning_rate or step_size
            beta: beta parameter for momentum (same as beta1 in Adam)
            beta2: beta2 parameter for Adam
            eps: epsilon parameter for Adam

        Outputs:
            No outputs
        Notes:
            self.parameters dictionary has the following keys and values
            {"W1":[..],"b1":[..],"W2":[..],"b2":[..],..., \
                                "bnorm_beta1":[..],"bnorm_gamma1":[..],"bnorm_beta2":[..],"bnorm_gamma2":[..],...}
            self.stats dictionary has the following keys and values

            {"VdW1":[..],"Vdb1":[..],"VdW2":[..],"Vdb2":[..],...} #velocity parameters
            {"GdW1":[..],"Gdb1":[..],"GdW2":[..],"Gdb2":[..],...} #Gradient-Squares parameters
            {"Vbnorm_beta1":[..],"Vbnorm_gamma1":[..],"Vbnorm_beta2":[..],\
                                "Vbnorm_gamma2":[..],...} #batchnorm velocity parameters
            {"Gbnorm_beta1":[..],"Gbnorm_gamma1":[..],"Gbnorm_beta2":[..],\
                                "Gbnorm_gamma2":[..],...} #batchnorm Gradient-Square parameters

            Bias correction is not required.
        '''

        gradients = dict(zip(self.parameters.keys(), gradients))

        for ly in range(self.num_layers):
            if self.apply_momentum:
                # Apply Adam momentum to parameters W's and b's.
                # You will need to update the Velocity parameters VdW's and Vdb's
                # self.stats["VdW" + str(ly+1)] =
                # self.stats["Vdb" + str(ly+1)] =
                # You will need to update the Gradient-Squares parameters GdW's and Gdb's
                # self.stats["GdW" + str(ly+1)] =
                # self.stats["Gdb" + str(ly+1)] =
                # You will need to update the parameters W's and b's
                # self.parameters["W" + str(ly+1)] =
                # self.parameters["b" + str(ly+1)] =
                # NOTE:
                # In the torch.no_grad() environment, parameter updates which are not in_place
                # may create new parameters with requires_grad=False.
                # autograd will not track gradients for the new parameters if requires_grad is False
                # Esnsure your self.parameters updates are inplace here
                # a += 1 is an inplace operation. It updates the current location of 'a'
                # a = a + 1 in NOT inplace. It creates a new location for 'a'
                ### BEGIN SOLUTION ###
                # update Velocities
                self.stats['VdW'+str(ly+1)] *= beta
                self.stats['VdW'+str(ly+1)] += (1-beta) * gradients['W'+str(ly+1)]
                self.stats['Vdb'+str(ly+1)] *= beta
                self.stats['Vdb'+str(ly+1)] += (1-beta) * gradients['b'+str(ly+1)]
                # update Gradient-Squares
                self.stats['GdW'+str(ly+1)] *= beta2
                self.stats['GdW'+str(ly+1)] += (1-beta2) * torch.square(gradients['W'+str(ly+1)])
                self.stats['Gdb'+str(ly+1)] *= beta2
                self.stats['Gdb'+str(ly+1)] += (1-beta2) * torch.square(gradients['b'+str(ly+1)])
                # update parameters
                self.parameters['W'+str(ly+1)] -= alpha / torch.sqrt(self.stats['GdW'+str(ly+1)] + eps) * self.stats['VdW'+str(ly+1)]
                self.parameters['b'+str(ly+1)] -= alpha / torch.sqrt(self.stats['Gdb'+str(ly+1)] + eps) * self.stats['Vdb'+str(ly+1)]
                ### END SOLUTION ###
            else:
                # When no momentum apply regular gradient descent
                self.parameters["W"+str(ly+1)] -= alpha * gradients["W"+str(ly+1)]
                self.parameters["b"+str(ly+1)] -= alpha * gradients["b"+str(ly+1)]

            # The Adam momentum for batch norm parameters has been implemented below
            if self.apply_momentum and self.bnorm_list[ly]:
                # update Velocities
                self.stats['Vbnorm_beta'+str(ly+1)] = beta*self.stats['Vbnorm_beta'+str(ly+1)] + \
                                                        (1 - beta)*gradients["bnorm_beta"+str(ly+1)]
                self.stats['Vbnorm_gamma'+str(ly+1)] = beta*self.stats['Vbnorm_gamma'+str(ly+1)] + \
                                                        (1 - beta)*gradients["bnorm_gamma"+str(ly+1)]
                # update Gradient-Squares
                self.stats['Gbnorm_beta'+str(ly+1)] = beta2*self.stats['Gbnorm_beta'+str(ly+1)] + \
                                                        (1 - beta2)*(gradients["bnorm_beta"+str(ly+1)]**2)
                self.stats['Gbnorm_gamma'+str(ly+1)] = beta2*self.stats['Gbnorm_gamma'+str(ly+1)] + \
                                                        (1 - beta2)*(gradients["bnorm_gamma"+str(ly+1)]**2)
                # update parameters
                self.parameters['bnorm_beta' + str(ly+1)] -= alpha*self.stats['Vbnorm_beta'+str(ly+1)]/\
                                                            torch.sqrt(self.stats['Gbnorm_beta'+str(ly+1)] + eps)
                self.parameters['bnorm_gamma' + str(ly+1)] -= alpha*self.stats['Vbnorm_gamma'+str(ly+1)]/\
                                                            torch.sqrt(self.stats['Gbnorm_gamma'+str(ly+1)] + eps)
            elif self.bnorm_list[ly]:
                # When no momentum apply regular gradient descent
                self.parameters['bnorm_beta' + str(ly+1)] -= alpha * gradients["bnorm_beta"+str(ly+1)]
                self.parameters['bnorm_gamma' + str(ly+1)] -= alpha * gradients["bnorm_gamma"+str(ly+1)]

def train(model, x, y, num_epochs=20, learning_rate=0.1, decay_rate=1, \
          minibatch_size=32, apply_momentum=False, seed=1, log=True, log_step=500):
    cost = []

    itr = 0
    torch.manual_seed(seed)
    print(num_epochs)
    for ii in range(num_epochs):

        start = timer()
        epoch_loss = 0
        # estimate stepsize alpha using (learning_rate*1/(1+decay_rate*epoch_number))
        ### BEGIN SOLUTION ###
        alpha = (learning_rate*1/(1+decay_rate*ii))
        ### END SOLUTION ###
        if log:
            print('------- Epoch {} -------'.format(ii+1))

        ## Shuffle data in epoch and create minibatches for minibatch gradient descent
        shuffle_index = torch.randperm(y.shape[0], dtype=torch.int)
        minibatches = torch.split(shuffle_index, minibatch_size)

        for minibatch_idx in minibatches:

            ## Forward Propagation and loss ##
            ## Estimate Softmax and Cross-entropy loss using F.cross_entropy()
            ## The F.cross_entropy() function does a Softmax before the cross-entropy
            ## logits = model.forward(..., mode='train')
            ## loss = F.cross_entropy(logits.T, y[minibatch_idx])
            ### BEGIN SOLUTION ###
            logits = model.forward(x[:, minibatch_idx], mode='train')
            loss = F.cross_entropy(logits.T, y[minibatch_idx])
            ### END SOLUTION ###

            ## Compute gradient and update parameters ##
            ## The grad(.) function gets the gradients of the loss w.r.t parameters in a list
            ## The output gradients is a list of corresponding parameter gradients
            gradients = grad(loss, list(model.parameters.values()))

            with torch.no_grad():
                if apply_momentum:
                    model.update_parameters_with_momentum_Adam(gradients, alpha)
                else:
                    model.update_parameters(gradients, alpha)

            epoch_loss += loss.item()
            if log and itr % log_step == 0:
                print("Minibatch loss at iteration %i is: %.05f, learning rate: %.05f" %(itr, loss.item(), alpha))
            itr+=1


        ## Print loss every epoch ##
        epoch_loss /= len(minibatches)
        print('Epoch: %02d | Loss: %.5f | Time: %.5f | TPTI: %.5f' % ((ii+1),epoch_loss, timer()-start, (timer()-start)/itr*1000))
        cost.append(epoch_loss)
    return cost

def evaluate(model, x, y=torch.tensor([]), mode='test', minibatch_size=32):
    '''
    function to evaluate a trained model
    
    Inputs:
        model: trained neural network
        x: data of the form (n,m) where m is the number of samples and (n,1) is the dimension of each sample
        y: (optional) m labels in 1D tensor 
        mode: (optional) 'train' or 'test'
        minibatch_size: (optional) 
        
    Outputs:
        ypred: tensor of predicted labels in 1D tensor
        loss: model loss (>0 when labels y are provided)
        acc: percentage accuracy (>0 when labels are provided)
    '''
    shuffle_index = torch.randperm(x.shape[1], dtype=torch.int)
    minibatches = torch.split(shuffle_index, minibatch_size)
    loss = -1
    acc = -1
    if y.numel() != 0:
        ypred = -1 + torch.zeros_like(y, dtype=torch.int64)
    else:
        ypred = -1 + torch.zeros((x.shape[1]), dtype=torch.int64)
    for minibatch_idx in minibatches:
        logits = model.forward(x[:,minibatch_idx], mode)
        ypred[minibatch_idx] = torch.argmax(logits,axis = 0)
        if y.numel() != 0:
            loss += F.cross_entropy(logits.T, y[minibatch_idx])
    loss /= len(minibatches)
    if y.numel() != 0:
        acc = torch.mean((ypred == y).float())*100
    return ypred, loss, acc
